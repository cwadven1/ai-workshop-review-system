ShopWeekReview 집계 DML 소스: reviewtime.Review  ·  BigQuery → import 후 집계  ·  idx_store_year_week 인덱스

seed_shop_week_review_init.sql  — 각 가게별 첫 12주치 초기 삽입 
-- ① 각 가게별 첫 리뷰 주차 기준 12주치 초기 데이터 삽입
INSERT INTO shop_week_review
    (shop_id, year, week_number, count, average,
     summary, positive_count, negative_count, neutral_count)

WITH
  -- 삭제·블라인드 제외한 유효 리뷰의 (vendor, ISO주차) 조합
  distinct_vendor_weeks AS (
    SELECT DISTINCT vendor_id, YEARWEEK(created_at, 3) AS yw
    FROM   review
    WHERE  deleted = 0 AND blinded = 0
  ),
  -- 각 가게별 ISO 주차 오름차순 순번 부여
  ranked_weeks AS (
    SELECT vendor_id, yw,
           ROW_NUMBER() OVER (PARTITION BY vendor_id ORDER BY yw) AS rn
    FROM   distinct_vendor_weeks
  ),
  -- 첫 12주만 추출
  first_12_weeks AS (
    SELECT vendor_id, yw FROM ranked_weeks WHERE rn <= 12
  )

SELECT
    r.vendor_id,
    FLOOR(f.yw / 100)        AS year,
    MOD(f.yw, 100)            AS week_number,
    COUNT(r.id)               AS count,
    ROUND(AVG(r.rating), 2)   AS average,
    NULL, NULL, NULL, NULL    -- summary / positive / negative / neutral_count
FROM   review r
JOIN   first_12_weeks f
  ON   r.vendor_id = f.vendor_id
  AND  YEARWEEK(r.created_at, 3) = f.yw
WHERE  r.deleted = 0 AND r.blinded = 0
GROUP  BY r.vendor_id, f.yw;
update_shop_week_review_incremental.sql  — 변경된 조합만 감지하여 재집계 (스케줄러 실행용) 
-- ② 변경된 (shop, year, week) 조합만 감지하여 증분 재집계
--    watermark = stores_shopweekreview.updated_at
--    변경 없으면 INSERT/UPDATE 0건 → 성능 영향 없음
INSERT INTO stores_shopweekreview (shop_id, year, week_number, count, average)

WITH
-- ① watermark 이후 변경된 리뷰가 있는 (shop, year, week)
--    created_at / updated_at / deleted_at / blinded_at 모두 감지
changed_combos AS (
    SELECT DISTINCT
        r.store_id,
        FLOOR(YEARWEEK(r.created_at, 3) / 100) AS year,
        MOD(YEARWEEK(r.created_at, 3), 100)    AS week_number
    FROM reviews_review r
    INNER JOIN stores_shopweekreview swr
           ON  swr.shop_id     = r.store_id
           AND swr.year        = FLOOR(YEARWEEK(r.created_at, 3) / 100)
           AND swr.week_number = MOD(YEARWEEK(r.created_at, 3), 100)
    WHERE r.created_at  > swr.updated_at
       OR r.updated_at  > swr.updated_at
       OR r.deleted_at  > swr.updated_at
       OR r.blinded_at  > swr.updated_at
),
-- ② 아직 집계 row 없는 신규 조합
new_combos AS (
    SELECT DISTINCT
        r.store_id,
        FLOOR(YEARWEEK(r.created_at, 3) / 100) AS year,
        MOD(YEARWEEK(r.created_at, 3), 100)    AS week_number
    FROM reviews_review r
    LEFT JOIN stores_shopweekreview swr
           ON  swr.shop_id     = r.store_id
           AND swr.year        = FLOOR(YEARWEEK(r.created_at, 3) / 100)
           AND swr.week_number = MOD(YEARWEEK(r.created_at, 3), 100)
    WHERE swr.shop_id IS NULL
),
target_combos AS (
    SELECT store_id, year, week_number FROM changed_combos
    UNION
    SELECT store_id, year, week_number FROM new_combos
)

-- ③ 현재 DB 상태 기준 활성 리뷰만 재집계
SELECT
    t.store_id             AS shop_id,
    t.year,
    t.week_number,
    COUNT(r.id)            AS count,
    ROUND(AVG(r.rating), 2) AS average
FROM target_combos t
JOIN reviews_review r
  ON  r.store_id                              = t.store_id
  AND FLOOR(YEARWEEK(r.created_at, 3) / 100) = t.year
  AND MOD(YEARWEEK(r.created_at, 3), 100)   = t.week_number
  AND r.is_deleted = FALSE
  AND r.is_blinded = FALSE
GROUP BY t.store_id, t.year, t.week_number

ON DUPLICATE KEY UPDATE
    count      = VALUES(count),
    average    = VALUES(average),
    updated_at = NOW();  -- watermark 갱신

ShopWeekReview 집계 — Django Command 소스: reviews.Review (reviewtime ORM)  ·  ISO 주차: year=2025, week_number=14

seed_shop_week_review_init.py  — 각 가게별 첫 12주치 초기 삽입 
# python manage.py seed_shop_week_review_init
from django.core.management.base import BaseCommand
from django.db.models import Avg, Count
from reviews.models import Review
from stores.models import ShopWeekReview   # 추가 예정 모델

class Command(BaseCommand):
    help = "각 가게별 첫 12주치 ShopWeekReview 초기 삽입"

    def handle(self, *args, **options):
        # 가게별 (year, week_number) 오름차순 distinct 목록
        date_qs = (
            Review.objects
            .values("store_id", "review_date")
            .distinct()
            .order_by("store_id", "review_date")
        )

        # 가게별 첫 12주만 추출
        store_weeks = {}
        for row in date_qs:
            sid = row["store_id"]
            iso = row["review_date"].isocalendar()
            yw = (int(iso[0]), int(iso[1]))
            if sid not in store_weeks:
                store_weeks[sid] = []
            if yw not in store_weeks[sid] and len(store_weeks[sid]) < 12:
                store_weeks[sid].append(yw)

        created = 0
        for store_id, year_weeks in store_weeks.items():
            for year, week_number in year_weeks:
                revs = Review.objects.filter(
                    store_id=store_id, review_date__week=week_number,
                    is_deleted=False, is_blinded=False,
                )
                ShopWeekReview.objects.create(
                    shop_id=store_id,
                    year=year, week_number=week_number,
                    count=revs.count(),
                    average=round(revs.aggregate(Avg("rating"))["rating__avg"] or 0, 2),
                )
                created += 1

        self.stdout.write(self.style.SUCCESS(f"총 {created}건 삽입 완료"))
seed_shop_week_review_today.py  — 오늘 기준 현재 주차 UPSERT (스케줄러 실행용) 
# python manage.py seed_shop_week_review_today
from datetime import date
from django.core.management.base import BaseCommand
from django.db.models import Avg, Count
from reviews.models import Review
from stores.models import ShopWeekReview   # 추가 예정 모델

class Command(BaseCommand):
    help = "오늘 기준 현재 주차 ShopWeekReview 집계 UPSERT (스케줄러용)"

    def handle(self, *args, **options):
        today     = date.today()
        iso       = today.isocalendar()
        year, week = int(iso[0]), int(iso[1])

        # 이번 주 리뷰 → 가게별 count / average
        agg = (
            Review.objects
            .filter(review_date__week=week, is_deleted=False, is_blinded=False)
            .values("store_id")
            .annotate(cnt=Count("id"), avg=Avg("rating"))
        )

        upserted = 0
        for row in agg:
            ShopWeekReview.objects.update_or_create(
                shop_id=row["store_id"],
                year=year, week_number=week,
                defaults={
                    "count":   row["cnt"],
                    "average": round(row["avg"], 2),
                },
            )
            upserted += 1

        self.stdout.write(
            self.style.SUCCESS(f"{year}-W{week:02d} | {upserted}개 가게 UPSERT 완료")
        )
