models.py + extract_keywords.py — Keyword 테이블 기반 upsert 
# ── Keyword 모델 ─────────────────────────────────────
class Keyword(models.Model):
    word      = models.CharField(max_length=50, unique=True)
    sentiment = models.CharField(choices=[               # positive / negative / neutral
        ("positive", "긍정"), ("negative", "부정"), ("neutral", "중립")
    ])

# ── 키워드 추출 (Gemini 또는 형태소) ───────────────────
def extract_keywords(review_text, rating):
    # 방법 A: Gemini API
    raw_words = gemini_extract(review_text)
    # 방법 B: kiwipiepy 형태소 분석 (로컬, API 비용 없음)
    # raw_words = kiwi_extract(review_text)

    sentiment = (
        "positive" if rating >= 4 else
        "negative" if rating <= 2 else "neutral"
    )

    keywords = []
    for word in raw_words:
        kw, _ = Keyword.objects.get_or_create(            # ③ 있으면 재사용 / 없으면 생성
            word=word,
            defaults={"sentiment": sentiment}
        )
        keywords.append(kw)

    return keywords                                          # ④ ReviewKeyword 테이블에 연결 저장

😊 Review.sentiment — 분류 예시
리뷰 내용 매칭 키워드 점수 결과
"맛있어요! 배달도 빠르고 포장도 깔끔해요" 맛있어요(+1) · 빠르고(+1) · 깔끔(+1) +3 positive
"음식이 신선하고 양도 많아서 만족해요" 신선(+1) · 만족(+1) +2 positive
"음식이 식어서 왔고 국물이 짰어요" 식어(-1) · 짰어(-1) -2 negative
"배달이 너무 늦고 포장이 엉망이에요" 늦고(-1) · 엉망(-1) -2 negative
"그냥 보통이에요. 특별한 건 없어요." 매칭 없음 0 neutral

키워드 집계 코드

generate_summaries.py + stores/views.py 
# ── 주차별 키워드 집계 (generate_summaries.py)
all_keywords = []
for r in reviews:
    all_keywords.extend(r.keywords)        # JSONField 배열 펼치기

keyword_counts = defaultdict(int)
for kw in all_keywords:
    keyword_counts[kw] += 1

# 상위 5개 → WeeklySummary.top_keywords 에 저장
top_keywords = sorted(keyword_counts.items(), key=lambda x: -x[1])[:5]
top_keywords = [{"keyword": kw, "count": cnt} for kw, cnt in top_keywords]


# ── 최근 4주 cross-week 재집계 (stores/views.py)
kw_counter = Counter()
for m in recent_weeks:              # 최근 4주 WeeklySummary
    for kw in (m['summary'].top_keywords or []):
        kw_counter[kw['keyword']] += kw.get('count', 1)

top_keywords = [
    {'keyword': k, 'count': c}
    for k, c in kw_counter.most_common(8)   # 상위 8개
]
